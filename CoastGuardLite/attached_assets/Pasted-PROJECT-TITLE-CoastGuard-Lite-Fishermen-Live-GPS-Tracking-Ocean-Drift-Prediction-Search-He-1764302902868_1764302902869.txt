const R = 6371; // Earth radius

function calculateDrift(lat, lng, windSpd, windDir, curSpd, curDir, hours) {
    // composite drift speed & direction
    const combinedSpeed = curSpd + (windSpd * 0.03);
    const angle = (curDir + (windDir * 0.2)) * Math.PI / 180;

    const distance = combinedSpeed * hours / 60; // km

    const newLat = lat + (distance / R) * (180 / Math.PI) * Math.cos(angle);
    const newLng = lng + (distance / R) * (180 / Math.PI) * Math.sin(angle);

    return { lat: newLat, lng: newLng };
}
